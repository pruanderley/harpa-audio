name: Build APK Harpa Crista

on:
  push:
    branches: [ main, master ]
  workflow_dispatch:

jobs:
  build-apk:
    runs-on: ubuntu-latest

    steps:
      - name: Checkout repositório
        uses: actions/checkout@v4

      - name: Configurar Java 17
        uses: actions/setup-java@v4
        with:
          java-version: '17'
          distribution: 'temurin'

      - name: Configurar Node.js
        uses: actions/setup-node@v4
        with:
          node-version: '22'

      - name: Instalar dependências
        run: npm install

      - name: Adicionar plataforma Android
        run: |
          rm -rf android
          npx cap add android

      - name: Injetar MainActivity no app
        run: |
          mkdir -p android/app/src/main/java/com/harpa/offline
          cp patches/MainActivity.java \
             android/app/src/main/java/com/harpa/offline/MainActivity.java

      - name: Copiar ícone para www/ e resources/
        run: |
          cp icon-192.png www/icon-192.png
          mkdir -p resources
          cp icon.png resources/icon.png
          npx @capacitor/assets generate --android

      - name: Corrigir tela branca
        run: |
          mkdir -p android/app/src/main/res/values
          mkdir -p android/app/src/main/res/drawable
          printf '%s' '<?xml version="1.0" encoding="utf-8"?><layer-list xmlns:android="http://schemas.android.com/apk/res/android"><item><color android:color="#14101d"/></item></layer-list>' > android/app/src/main/res/drawable/launch_background.xml
          printf '%s' '<?xml version="1.0" encoding="utf-8"?><resources><style name="AppTheme" parent="Theme.AppCompat.Light.NoActionBar"><item name="colorPrimary">#8E6FD1</item><item name="colorPrimaryDark">#6a3fc2</item><item name="colorAccent">#b58cf0</item></style><style name="AppTheme.NoActionBarLaunch" parent="AppTheme"><item name="android:windowBackground">@drawable/launch_background</item><item name="android:windowNoTitle">true</item><item name="android:windowFullscreen">false</item></style></resources>' > android/app/src/main/res/values/styles.xml

      - name: Sincronizar arquivos web
        run: npx cap sync android

      - name: Compilar APK
        run: |
          cd android
          chmod +x gradlew
          ./gradlew assembleDebug --no-daemon

      - name: Publicar APK
        uses: actions/upload-artifact@v4
        with:
          name: Harpa-Crista-Offline-APK
          path: android/app/build/outputs/apk/debug/app-debug.apk
          retention-days: 30
